public class LireFichier {
    public static void main(String[] args) {
        try {
            File file = new File("sortie.txt");
            Scanner scanner = new Scanner(file);
            while (scanner.hasNextLine()) {
                System.out.println(scanner.nextLine());
            }
            scanner.close();
        } catch (Exception e) {
