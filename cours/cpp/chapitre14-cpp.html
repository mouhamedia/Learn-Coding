Chapitre 14 : Polymorphisme ⚙️

Permets à des objets de se comporter différemment.

🔹 Qu’est-ce que le polymorphisme ?

Le polymorphisme permet à une même méthode d’avoir des comportements différents selon l’objet.

📌 Méthodes virtuelles

Utilise virtual pour permettre à une sous-classe de redéfinir une méthode.


#include 
using namespace std;

class Animal {
public:
    virtual void faireSon() {
        cout << "Son d'animal" << endl;
    }
};

class Chat : public Animal {
public:
    void faireSon() override {
        cout << "Miaou !" << endl;
    }
};

int main() {
    Animal* animal = new Chat();
    animal->faireSon();
    delete animal;
    return 0;
}

💻 Surcharge de méthodes

La surcharge utilise le même nom de méthode avec des paramètres différents.


#include 
using namespace std;

class Calcul {
public:
    int addition(int a, int b) {
        return a + b;
    }
    double addition(double a, double b) {
        return a + b;
    }
};

int main() {
    Calcul c;
    cout << c.addition(5, 3) << endl;
    cout << c.addition(5.5, 3.2) << endl;
    return 0;
}

🛠 Exercice pratique

Crée une classe Forme avec une méthode virtuelle aire, redéfinie dans une classe Cercle.


#include 
using namespace std;

class Forme {
public:
    virtual double aire() {
        return 0.0;
    }
};

class Cercle : public Forme {
private:
    double rayon;
public:
    Cercle(double r) : rayon(r) {}
    double aire() override {
        return 3.14 * rayon * rayon;
    }
};

int main() {
    Forme* forme = new Cercle(5);
    cout << "Aire : " << forme->aire() << endl;
    delete forme;
    return 0;
}
