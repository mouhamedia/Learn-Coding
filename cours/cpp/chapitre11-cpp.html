Chapitre 11 : Constructeurs et destructeurs ⚙️

Initialise et nettoie tes objets automatiquement.

🔹 Qu’est-ce qu’un constructeur ?

Un constructeur est une fonction spéciale qui s’exécute quand un objet est créé, pour lui donner des valeurs initiales.


#include 
#include 
using namespace std;

class Voiture {
public:
    string marque;
    Voiture(string m) {
        marque = m;
        cout << "Voiture créée : " << marque << endl;
    }
};

int main() {
    Voiture v("Toyota");
    return 0;
}

📌 Destructeurs

Un destructeur s’exécute quand un objet est détruit, pour libérer des ressources.


#include 
#include 
using namespace std;

class Voiture {
public:
    string marque;
    Voiture(string m) {
        marque = m;
        cout << "Voiture créée : " << marque << endl;
    }
    ~Voiture() {
        cout << "Voiture détruite : " << marque << endl;
    }
};

int main() {
    Voiture v("Honda");
    return 0;
}

🛠 Exercice pratique

Crée une classe Personne avec un constructeur qui initialise le nom et un destructeur qui affiche un message.


#include 
#include 
using namespace std;

class Personne {
public:
    string nom;
    Personne(string n) {
        nom = n;
        cout << "Personne créée : " << nom << endl;
    }
    ~Personne() {
        cout << "Personne détruite : " << nom << endl;
    }
};

int main() {
    Personne p("Alice");
    return 0;
}
