Chapitre 13 : Héritage ⚙️

Partage des caractéristiques entre classes.

🔹 Qu’est-ce que l’héritage ?

L’héritage permet à une classe (fille) de réutiliser le code d’une autre classe (mère).

📌 Utiliser l’héritage

Utilise : public pour hériter d’une classe.


#include 
#include 
using namespace std;

class Vehicule {
public:
    string marque;
    Vehicule(string m) {
        marque = m;
    }
    void afficher() {
        cout << "Marque : " << marque << endl;
    }
};

class Voiture : public Vehicule {
public:
    Voiture(string m) : Vehicule(m) {}
};

int main() {
    Voiture v("Toyota");
    v.afficher();
    return 0;
}

💻 Ajouter des attributs

La classe fille peut ajouter ses propres attributs et méthodes.


#include 
#include 
using namespace std;

class Vehicule {
public:
    string marque;
    Vehicule(string m) {
        marque = m;
    }
};

class Moto : public Vehicule {
public:
    int cylindree;
    Moto(string m, int c) : Vehicule(m) {
        cylindree = c;
    }
    void afficher() {
        cout << "Moto : " << marque << ", Cylindrée : " << cylindree << endl;
    }
};

int main() {
    Moto m("Yamaha", 600);
    m.afficher();
    return 0;
}

🛠 Exercice pratique

Crée une classe Animal et une classe Chien qui hérite de ses attributs.


#include 
#include 
using namespace std;

class Animal {
public:
    string nom;
    Animal(string n) {
        nom = n;
    }
};

class Chien : public Animal {
public:
    Chien(string n) : Animal(n) {}
    void aboyer() {
        cout << nom << " dit : Wouf !" << endl;
    }
};

int main() {
    Chien c("Rex");
    c.aboyer();
    return 0;
}
