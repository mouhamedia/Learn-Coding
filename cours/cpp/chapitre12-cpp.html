Chapitre 12 : Encapsulation ⚙️

Protège les données de tes objets avec l’encapsulation.

🔹 Qu’est-ce que l’encapsulation ?

L’encapsulation consiste à cacher les données d’un objet et à ne les modifier qu’avec des méthodes spéciales.

📌 Attributs privés

Utilise private pour empêcher l’accès direct aux attributs.


#include 
using namespace std;

class Compte {
private:
    double solde;

public:
    Compte(double s) {
        solde = s;
    }
};

💻 Getters et setters

Les getters récupèrent une valeur, les setters la modifient.


#include 
using namespace std;

class Compte {
private:
    double solde;

public:
    Compte(double s) {
        solde = s;
    }
    double getSolde() {
        return solde;
    }
    void setSolde(double s) {
        if (s >= 0) {
            solde = s;
        }
    }
};

int main() {
    Compte c(100.0);
    c.setSolde(150.0);
    cout << "Solde : " << c.getSolde() << endl;
    return 0;
}

🛠 Exercice pratique

Crée une classe Etudiant avec un attribut privé note et des getters/setters.


#include 
using namespace std;

class Etudiant {
private:
    int note;

public:
    Etudiant(int n) {
        note = n;
    }
    int getNote() {
        return note;
    }
    void setNote(int n) {
        if (n >= 0 && n <= 20) {
            note = n;
        }
    }
};

int main() {
    Etudiant e(15);
    e.setNote(18);
    cout << "Note : " << e.getNote() << endl;
    return 0;
}
